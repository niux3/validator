let options = {
    fields : {
        'query': {
            "notempty" : {
                "error" : "ce champs doit contenir un terme de recherche"
            }
        }
    }
};
let validate = new Validator(options);
validate.form();

let listRequireField = [
    'input[type=text]',
    'input[type=date]',
    'input[type=password]',
    'input[type=radio]',
    'input[type=checkbox]',
    'select',
    'textarea',
];
document.querySelectorAll(listRequireField.join(',')).forEach(($field)=>{
    $field.addEventListener('blur', (e)=>{
        validate.element(e.target);
    });
});
